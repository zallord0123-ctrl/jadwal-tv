var NAMA_SHEET_TARGET = "Jadwal TV"; function onOpen() { SpreadsheetApp.getUi().createMenu('📺 UPDATE TV') .addItem('Kirim Sheet Ini ke TV', 'otomasiPindahJadwal') .addToUi(); } function otomasiPindahJadwal() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sourceSheet = ss.getActiveSheet(); var targetSheet = ss.getSheetByName(NAMA_SHEET_TARGET); if (!targetSheet) { SpreadsheetApp.getUi().alert("Sheet '" + NAMA_SHEET_TARGET + "' tidak ditemukan!"); return; } var data = sourceSheet.getDataRange().getDisplayValues(); var result = []; // Looping membaca seluruh baris dari atas ke bawah for (var i = 0; i < data.length; i++) { var colA = String(data[i][0]).trim().toLowerCase(); var colB = String(data[i][1]).trim().toLowerCase(); // 1. CARI BARIS JANGKAR: "Hari/Sesi" if (colA.includes("hari/sesi") || colB.includes("hari/sesi") || colA.includes("hari / sesi") || colB.includes("hari / sesi")) { var sesiTime = null; for (var c = 0; c < data[i].length; c++) { var match = String(data[i][c]).match(/\d{2}\.\d{2}/); if (match) { sesiTime = match[0]; break; } } if (!sesiTime) continue; // 2. SCAN KE BAWAH (Mencari Ruang, Mapel, dan Guru milik sesi tersebut) var roomRow = -1, mapelRow = -1, guruRow = -1; // Scan ke bawah mencari baris "Sesi / Rg" for (var down = i; down <= Math.min(i + 8, data.length - 1); down++) { var label = String(data[down][1]).trim().toLowerCase(); if (label.includes("sesi / rg") || label.includes("sesi/rg") || label.includes("sesi/ruang")) { roomRow = down; break; } } if (roomRow === -1) continue; // Scan ke bawah mencari baris Mapel dan Guru for (var down = roomRow + 1; down <= Math.min(roomRow + 25, data.length - 1); down++) { var label = String(data[down][1]).trim().toLowerCase(); if (label.includes("b.studi/kls") || label.includes("b.studi / kls") || label.includes("b. studi")) { mapelRow = down; } else if (label.includes("guru tetap") || label.includes("guru")) { guruRow = down; break; } } if (mapelRow === -1 || guruRow === -1) continue; // Batas baris nama siswa var studentStart = roomRow + 1; var studentEnd = mapelRow - 1; // 3. AMBIL SEMUA KOLOM (Tanpa Batas Kaku, otomatis sampai Ruang Komik) for (var c = 2; c < data[roomRow].length; c++) { var roomName = String(data[roomRow][c]).trim(); if (!roomName || roomName === "-") continue; var guruName = String(data[guruRow][c]).trim(); // PENTING: Jangan masukkan ke TV jika Guru tidak ada atau tertulis "Kosong" if (!guruName || guruName === "-" || guruName.toLowerCase() === "kosong") continue; var mapelName = String(data[mapelRow][c]).trim(); // Gabungkan nama siswa di ruangan tersebut var students = []; for (var r = studentStart; r <= studentEnd; r++) { var sName = String(data[r][c]).trim(); if (sName && sName !== "-") { students.push(sName); } } result.push([sesiTime, guruName, roomName, mapelName, students.join("\n")]); } } } // 4. BERSIHKAN & MASUKKAN DATA KE "Jadwal TV" targetSheet.clearContents(); targetSheet.appendRow(["sesi", "guru", "ruang", "mapel", "siswa"]); if (result.length > 0) { targetSheet.getRange(2, 1, result.length, 5).setValues(result); } SpreadsheetApp.getUi().alert("✅ UPDATE BERHASIL!\n\nJadwal sudah kembali sinkron secara presisi.\nRuang 14 & Ruang Komik dipastikan masuk!"); }
